ckday = crime.pivot_table('發生地點', index=['month', 'day'], columns='案類',aggfunc='count').fillna(0)

ckday.index = [pd.datetime(2017, month, day) for (month, day) in ckday.index]
ckday = ckday/ckday.sum()*100
